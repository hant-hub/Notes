                \begin{gather}
                    Pr(A) = \frac{7}{10}\\

                    Pr(C) = \frac{1}{10}\\
                    Pr(R | A) = \frac{20}{30}\\
                    Pr(R | B) = \frac{10}{30}\\
                    Pr(R | C) = \frac{15}{30}\\
                    Pr(R) = Pr(R|A)Pr(A) + Pr(R|B)Pr(B) + Pr(R|C)Pr(C)\\
                    = \frac{35}{60}
                \end{gather}
            \item Chance of ultimate victory

                \begin{gather}
                   Pr(W) = 0.6\\ 
                   Pr(M) = 0.3\\
                   Pr(P) = 0.4\\
                   Pr(W \cap M) = 0.26\\
                   Pr(M \cap P) = 0.19\\
                   Pr(W \cap P) = 0.37\\
                   Pr(W \cap M \cap P) = 0.01\\
                   Pr(W \cup M \cup P)\\
                   = Pr(W) + Pr(M) + Pr(P) - Pr(W \cap M)
                   - Pr(W \cap P) - Pr(M \cap P) + Pr(M \cap W \cap P)\\
                   = 0.49
                \end{gather}
